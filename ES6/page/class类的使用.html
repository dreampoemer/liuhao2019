class Coder{
                    name(val){
                        console.log(val);
                        return val;
                    }
                    skill(val){
                        console.log(this.name('jspang')+':'+'Skill:'+val);
                    }

                    //在类的参数传递中我们用constructor( )进行传参。传递参数后可以直接使用this.xxx进行调用
                    constructor(a,b){
                        this.a=a;
                        this.b=b;
                    }
                 
                    add(){
                        return this.a+this.b;
                    }
                }
                 
                let jspang=new Coder(1,2);
                console.log(jspang.add());
