1. 字符串拼接 
                    
                    ES5：
                    let name = '李白';
                    let message = '欢迎'+name+'来到ES6 world~';
                    document.write(message);

                    ES6:
                    let name = '李白';
                    let message = '欢迎${name}来到ES6 world~';
                    document.write(message);

                    同理，ES6对运算的支持。
                    let a=1;
                    let b=2;
                    let result=`${a+b}`;
                    document.write(result);
